Configuração CORS 

// app.ts
app.use('*', cors({
  origin: [
    env.FRONTEND_URL,              // frontend em produção
    'http://localhost:5174'        // dev local
  ],
  credentials: true,
  allowHeaders: ['Content-Type', 'Authorization'],
  exposeHeaders: ['X-Workspace-RateLimit-Remaining']
}));

// Middleware protege todas as rotas
